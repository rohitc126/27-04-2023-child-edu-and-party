ALTER PROC [iFMS].[USP_INSERT_CHILD_EDUCATION_EXCEPTION_ENTRY]
(	
     @ERRORSTR				VARCHAR(200) OUTPUT,
     @CEA_EXP_ID            numeric(18, 0) OUTPUT,
	 @CEA_EXP_EMP_ID          int,
	 @CEA_EXP_BR_ID           int,
	 @CEA_EXP_CBS_DATE        varchar(10),
	 @CEA_EXP_AMT             numeric(18, 2) ,
	 @CEA_EXP_ADD_BY          numeric(12, 0),
	 @CEA_EXP_REMARKS varchar(200),
	 @CEA_EXP_EMP_NAME VArchar(100)

     )
AS
BEGIN
SET NOCOUNT ON;
	SET @ERRORSTR=''
	SET @CEA_EXP_ID=0

	DECLARE @CEA_EXP_APP_STATUS INT,@CNT INT
	SELECT @CEA_EXP_APP_STATUS = CEA_EXP_APP_STATUS,@CNT = 1 FROM [iFMS].[TBL_CHILD_EDU_EXCEPTION_ALLOW] WITH(NOLOCK) 
	WHERE CEA_EXP_EMP_ID=@CEA_EXP_EMP_ID AND CEA_EXP_CBS_DATE = CONVERT(DATETIME,@CEA_EXP_CBS_DATE,103) AND CEA_EXP_STATUS=1

	IF(ISNULL(@CNT,0) > 0)
	BEGIN 
		IF ISNULL(@CEA_EXP_APP_STATUS,0) = 0
		BEGIN
			SET @ERRORSTR='EMPLOYEE : '+@CEA_EXP_EMP_NAME + ' ALREADY REQUESTED AND PENDING FOR APPROVAL.'
			SET @CEA_EXP_ID =-1
			RETURN
		END  
		IF ISNULL(@CEA_EXP_APP_STATUS,0) = 1
		BEGIN
			SET @ERRORSTR='REQUESTED EMPLOYEE : ' +@CEA_EXP_EMP_NAME + ' IS APPROVED.'
			SET @CEA_EXP_ID =-1
			RETURN
		END 
		IF ISNULL(@CEA_EXP_APP_STATUS,0) = 2
		BEGIN
			SET @ERRORSTR='REQUESTED EMPLOYEE : ' + @CEA_EXP_EMP_NAME  + ' IS REJECTED.'
			SET @CEA_EXP_ID = -1
			RETURN
		END 
	 END
	 ELSE
      BEGIN	
			INSERT INTO [iFMS].[TBL_CHILD_EDU_EXCEPTION_ALLOW]
			(
				CEA_EXP_EMP_ID,         
				CEA_EXP_BR_ID,           
				CEA_EXP_CBS_DATE,        
				CEA_EXP_AMT,
				CEA_EXP_ADD_BY ,         
				CEA_EXP_REMARKS
			)
			VALUES
			(       
				@CEA_EXP_EMP_ID,          
				@CEA_EXP_BR_ID,          
			    CONVERT(DATETIME,@CEA_EXP_CBS_DATE,103),       
				@CEA_EXP_AMT, 
				@CEA_EXP_ADD_BY ,
				@CEA_EXP_REMARKS  
			)			
		SET @CEA_EXP_ID = (SELECT SCOPE_IDENTITY())		
		SET @ERRORSTR = ''			
    END
 
	IF @@ERROR <> 0
	BEGIN
		SET @ERRORSTR = 'DATA BASE ERROR OCCUR FOR EMPLOYEE CHILD EDUCATION EXCEPTION ENTRY!'		 	
		SET @CEA_EXP_ID = -1		 	
		RETURN
	END
END 

GO
